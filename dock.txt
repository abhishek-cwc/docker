Container:

Run Container:
	docker container run -d ubuntu
Enter inside container:
	docker container run -it ubuntu /bin/bash
Enter inside running container:
	docker exec -it <cid> bash
Inspect Container:	
	docker container inspect <cid>
Port Mapping
	docker container run -d -p 3600:80 nginx
Export Container:
	docker container export <cid> > abhi_nginx.tar
	
Import Container:
	docker image import abhi_nginx.tar abhi_nginx
	docker run -d -p 3605:80 abhi_nginx nginx -g "daemon off;"
	
Create image from container	
	docker container commit --author "abhishek pathak" -m "first commit" e544d940eba8 abhi_nginx
	docker container run -d -p 3602:80 abhi_nginx

Push image on dockerhub
	docker image tag abhi_nginx abhi86pathak/abhi_nginx:v2
	docker push abhi86pathak/abhi_nginx_git
	

Docker compose
	docker-compose up -d
	docker-compose down
	docker-compose logs webapp

Scale the container:
	docker compose up --scale webapp=2 -d
	
Rebuild docker image:	
	docker-compose down -v
	docker-compose build --no-cache
	docker-compose up -d
	docker ps
